1. Diagonal (right) Pattern 

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner a = new Scanner (System.in);
		int n=a.nextInt();
		int st=n;
		for(int i=1;i<=n;i++){
		    for(int j=1;j<=st;j++){
		        if(i==j)
		            System.out.print("* ");
		        else 
		            System.out.print(" ");
		    }System.out.println();
		}
	}
}

10. Swastik pattern 

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner a = new Scanner (System.in);
		int n=a.nextInt();
		int st=n;
		for(int i=1;i<=n;i++){
		    for(int j=1;j<=st;j++){
		        if(i==n/2+1 || j==n/2+1  || (i==1 && j>=n/2+1)||(i==n && j<=n/2+1)||(j==1 && i<=n/2+1)||(j==n && i>=n/2+1))
		            System.out.print("* ");
		        else 
		            System.out.print("  ");
		    }
		    System.out.println();
		}
	}
}


11. Name pattern (inc order)

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner a = new Scanner (System.in);
		String str=a.nextLine();
		int n=str.length();
		int st=1;
		for(int i=1;i<=n;i++){
		    for(int j=1;j<=st;j++){
		            System.out.print(str.charAt(j-1)+" ");
		    }
		    System.out.println();
		    st++;
		}
	}
}

15. Border Name pattern (zoho)

import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner a = new Scanner (System.in);
		String str=a.nextLine();
		int n=str.length();
		int st=n;
		for(int i=1;i<=n;i++){
		    for(int j=1;j<=st;j++){
		        if(i==1){
		            System.out.print(str.charAt(j-1)+" ");
		        }
		        else if(i==n)
		            System.out.print(str.charAt(n-j)+" ");
		        else if(j==1)
		            System.out.print(str.charAt(i-1)+" ");
		        else if(j==st){
		            System.out.print(str.charAt(n-i)+" ");}
		        else 
		            System.out.print("  ");
		        
		        
		    }
		    System.out.println();
		    
		}
	}
}
